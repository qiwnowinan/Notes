# 抽象类

有定义规范的作用

```py
from abc import ABC,abstractmethod
class Action(ABC):
    @abstractmethod # 抽象方法，子类必须实现，定义规范
    def execute(self):
        pass

action = Action() # 报错，有抽象方法，无法实例化

class SA(Action):
    def execute(self): # 实现抽象方法，否则无法实例化
        print("执行SA的execute方法")

def execute_action(action: Action): # 接受一个Action子类的参数，调用execute方法
    action.execute()
