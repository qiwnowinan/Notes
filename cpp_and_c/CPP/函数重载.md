# 函数重载

```cpp
int myMax(int a,int b);
double myMax(double a,double b);

myMax(12, 23);
myMax(15.5,12.5);
```

## 函数隐藏

```cpp
#include <iostream>
using namespace std;

class A {
public:
    A(int ii):i(ii) { cout << "A::A()" << endl; }
    ~A() { cout << "A::~A()" << endl; }
    void print() { cout << "A::print()" << i << endl; }
    void print(int i) { cout << "A::print()" << i << endl; } // 重载函数
private:
    int i;
};

class B : public A {
public:
    // 构造函数：调用 A 的构造函数传入 15
    B() : A(15) { cout << "B::B()" << endl; }

    // 析构函数
    ~B() { cout << "B::~B()" << endl; }

    // 新增的 print 函数
    void print() { cout << "B::print()" << endl; }
};

int main() {
    B b;   // 创建对象
    b.print(); // 调用的是B的print函数,A的全部print函数已经隐藏了
    return 0;
}
```
